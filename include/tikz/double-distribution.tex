
\documentclass[tikz,border=10pt]{standalone}

\begin{document}

\begin{tikzpicture}[scale=1.0]

	\def\A{4}
	\def\B{2}
	\def\C{3}
	\def\D{1}

	\coordinate (A) at (0,0);
	\coordinate (B) at (6,0);
	\coordinate (C) at (6,4);
	\coordinate (D) at (0,4);

	\coordinate (E) at (4,0);
	\coordinate (F) at (6,3);
	\coordinate (G) at (4,4);
	\coordinate (H) at (0,3);

	\draw (0,0) rectangle (6,4);
	\draw (E) -- (G);
	\draw (F) -- (H);

	\node[below] at ($(A)!0.5!(E)$) {$a$};
	\node[below] at ($(E)!0.5!(B)$) {$b$};
	\node[left] at ($(A)!0.5!(H)$) {$c$};
	\node[left] at ($(H)!0.5!(D)$) {$d$};

	\node at ({\A/2},{\C/2}) {$a \ c$};
	\node at ({\A + \B/2},{\C/2}) {$b \ c$};
	\node at ({\A/2},{\C + \D/2}) {$a \ d$};
	\node at ({\A + \B/2},{\C + \D/2}) {$b \ d$};

\end{tikzpicture}
