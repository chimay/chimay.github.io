% Minimal TikZ standalone example
\documentclass[tikz, border=1mm]{standalone}

\usepackage{amsmath}
\usepackage{tikz}

\usetikzlibrary{calc,angles,quotes}

\begin{document}

\begin{tikzpicture}[scale=0.7]

	\coordinate (P_1) at (0,0);
	\coordinate (P_2) at (4,0);
	\coordinate (P_3) at (7,3);
	\coordinate (P_4) at (5,6);
	\coordinate (P_5) at (3,7);
	\coordinate (P_6) at (-1,5);

	\draw (P_1) \foreach \i in {2,...,6} { -- (P_\i) } -- cycle;

	\draw[dashed] (P_1) -- (P_3);
	\draw[dashed] (P_1) -- (P_4);
	\draw[dashed] (P_1) -- (P_5);

	\foreach \i in {1,...,6} { \fill (P_\i) circle (0.7mm); }

	 \node[below] at (P_1) {$P_1$};
	 \node[below] at (P_2) {$P_2$};
	 \node[right] at (P_3) {$P_3$};
	 \node[above right] at (P_4) {$P_4$};
	 \node[above] at (P_5) {$P_5$};
	 \node[above left] at (P_6) {$P_6$};

\end{tikzpicture}
\end{document}
