	\coordinate (v1) at ($ (A) - (O) $);
	\coordinate (v2) at ($ (B) - (O) $);

	\coordinate (u2) at ($ (0,0)!1cm!(v2) $);
	\coordinate (u3) at ($ (0,0)!1cm!(v3) $);

	\coordinate (X) at ($ (A) + {1.5*\extlen}*(u1) $);
	\coordinate (Y) at ($ (B) + {1.5*\extlen}*(u2) $);

	\fill (B) circle (0.5mm);
	\fill (D) circle (0.5mm);
	\fill (E) circle (0.5mm);
	\fill (I) circle (0.5mm);
