% document %{{{1

\begin{document}
\begin{tikzpicture}[scale=1.0]

	% ---- parameters

	\def\len{6}
	\def\radius{0.65*\len}

	% ---- coordinates

	\coordinate (A) at (0,0);
	\coordinate (B) at (\len,0);

	\coordinate (Ra) at ($(A) + (\radius, 0)$);
	\coordinate (Rb) at ($(B) + (\radius, 0)$);

	% ---- intersections

	\tkzInterCC(A,Ra)(B,Rb)\tkzGetPoints{D}{E}

	\coordinate (I) at ($(D) + (0, 1)$);
	\coordinate (J) at ($(E) - (0, 1)$);

	% ---- segments

	\draw (A) -- (B);
	\draw (I) -- (J);

	% ---- circular arcs

	\draw (A) ++(-65:\radius) arc (-65:65:\radius);
	\draw (B) ++(115:\radius) arc (115:245:\radius);

	% ---- thick vertices

	\fill (A) circle (0.5mm);
	\fill (B) circle (0.5mm);
	\fill (D) circle (0.5mm);
	\fill (E) circle (0.5mm);

	% ---- vertices labels

	\node[below] at (A) {$A$};
	\node[below] at (B) {$B$};
	\node[label={[label distance=0.5]left:$D$}] at (D) {};
	\node[label={[label distance=0.5]left:$E$}] at (E) {};

	% ---- circular arcs labels

	\node at ($(20:{\radius+0.3}) + (0, 0)$) {$\mathscr{C}_1$};
	\node at ($(160:{\radius+0.3}) + (\len, 0)$) {$\mathscr{C}_2$};

\end{tikzpicture}
\end{document}
