% Minimal TikZ standalone example
\documentclass[tikz, border=1mm]{standalone}

\usepackage{amsmath}
\usepackage{tikz}

\usetikzlibrary{calc,angles,quotes}

\begin{document}

\begin{tikzpicture}[scale=1.2]

	% coordinate

	\coordinate (A) at (0,0);
	\coordinate (C) at (5,0);
	\coordinate (B) at (3,3);

	% feet of altitude from B to AC

	\coordinate (D) at (3,0);

	% triangle sides

	\draw (A) -- (B) -- (C) -- cycle;

	% altitude

	\draw[dashed] (B) -- (D);

	% right angle marker

	\pic [draw, angle radius=7pt, angle eccentricity=1]
	{right angle = B--D--A};

	% side labels

	\node[above left]  at ($(A)!0.5!(B)$) {$a$};
	\node[below] at ($(A)!0.5!(C)$) {$b$};
	\node[above right] at ($(B)!0.5!(C)$) {$c$};

	% angle gamma

	\pic[draw, ->, "$\gamma$", angle eccentricity=1.3, angle radius=1cm]
	{angle = C--A--B};

	% segment labels on altitude projection

	\node[above] at ($(A)!0.7!(D)$) {$u$};
	\node[above] at ($(D)!0.5!(C)$) {$b-u$};

	% height label

	\node[right] at ($(B)!0.5!(D)$) {$h$};

\end{tikzpicture}

\end{document}
