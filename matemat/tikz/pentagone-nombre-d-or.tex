% opening %{{{1

\begin{document}
\begin{tikzpicture}[scale=1.0]

% parameters %{{{1

	\tikzmath{
		\radius = 5.0 ;
		\rotation = 90 ;
		\numsidesfloat = 5.0 ;
		\numsides = int(\numsidesfloat) ;
		int \i ;
		for \i in {1,...,\numsides} {
			\ang{\i} = \rotation + 360 / \numsidesfloat * (\i-1) ;
		};
	}

% coordinates %{{{1

	\coordinate (O) at (0,0);

	\foreach \i in {1,...,\numsides}
		\coordinate (P_\i) at (\ang{\i}:\radius);

% intersections %{{{1

	\begin{pgfinterruptboundingbox}
		\foreach \i
			[
				evaluate=\i as \j using { int(mod(\i, \numsides) + 1) },
				evaluate=\i as \k using { int(mod(\i + 1, \numsides) + 1) },
			]
			in {1,...,\numsides}
		{
			\tkzInterLL(O,P_\j)(P_\i,P_\k)\tkzGetPoint{I_\j}
		}
	\end{pgfinterruptboundingbox}

	\begin{pgfinterruptboundingbox}
		\foreach \i
			[
				evaluate=\i as \j using { int(mod(\i, \numsides) + 1) },
				evaluate=\i as \k using { int(mod(\i + 1, \numsides) + 1) },
				evaluate=\i as \l using { int(mod(\i + 2, \numsides) + 1) }
			]
			in {1,...,\numsides}
		{
			\tkzInterLL(P_\i,P_\k)(P_\j,P_\l)\tkzGetPoint{S_\j}
		}
	\end{pgfinterruptboundingbox}

% points, dots, vertices %{{{1

	\fill (O) circle (0.4mm);

	\foreach \i in {1,...,\numsides}
		\fill (P_\i) circle (0.4mm);

	\foreach \i in {1,...,\numsides}
		\fill (S_\i) circle (0.4mm);

% segments, sides, lines %{{{1

	\draw (P_1) \foreach \i in {2,...,\numsides} { -- (P_\i) } -- cycle;

	\draw (P_1) -- (P_4);
	\draw (P_2) -- (P_5);

	\draw[dashdotdotted] (P_1) -- (P_3) -- (P_5);
	\draw[dashdotdotted] (P_2) -- (P_4);

% circles %{{{1

	\draw (O) circle (\radius);

% points, dots, vertices labels %{{{1

	\node[anchor=east] at ($ (O) + (220:0pt) $) {$O$};

	\node[above] at (P_1) {$P_1$};
	\node[left] at (P_2) {$P_2$};
	\node[below left] at (P_3) {$P_3$};
	\node[below right] at (P_4) {$P_4$};
	\node[right] at (P_5) {$P_5$};

	\node[above left] at (S_1) {$S_1$};
	\node[left] at (S_2) {$S_2$};
	\node[below] at (S_3) {$S_3$};
	\node[right] at (S_4) {$S_4$};
	\node[above right] at (S_5) {$S_5$};

% segments, sides, lines labels %{{{1

	\node[above left] at ($ (P_1)!0.5!(P_2) $) {$c$};
	\node[above right] at ($ (P_1)!0.5!(P_5) $) {$c$};
	\node[below right] at ($ (P_5)!0.5!(P_4) $) {$c$};

	\node[below] at ($ (P_2)!0.6!(S_1) $) {$a$};
	\node[below] at ($ (P_5)!0.6!(S_5) $) {$a$};
	\node[above right] at ($ (P_1)!0.6!(S_5) $) {$a$};
	\node[above right] at ($ (P_4)!0.6!(S_4) $) {$a$};

	\node[below] at ($ (S_1)!0.5!(S_5) $) {$b$};
	\node[above right] at ($ (S_4)!0.5!(S_5) $) {$b$};

% segments, sides, lines marks %{{{1

	\foreach \i
		[
			evaluate=\i as \j using { int(mod(\i,\numsides) + 1) }
		]
		in {1,...,\numsides}
	{
		\tkzMarkSegments[mark=||, size=2pt, pos=0.45](P_\i,P_\j);
		\tkzMarkSegments[mark=||, size=2pt, pos=0.55](P_\i,P_\j);
	}

	\foreach \i in {1,...,\numsides}
		\tkzMarkSegments[mark=||, size=2pt, pos=0.6](P_\i,S_\i);

	\foreach \i
		[
			evaluate=\i as \j using { int(mod(\i,\numsides) + 1) }
		]
		in {1,...,\numsides}
	{
		\tkzMarkSegments[mark=||, size=2pt, pos=0.6](P_\j,S_\i);
		\tkzMarkSegments[mark=|, size=2pt, pos=0.5](S_\i,S_\j);
	}

% circles labels %{{{1

	\node at ($ (O) + (60:\radius + 0.3) $) {$\mathscr{C}$};

% angles labels %{{{1

	\pic[draw, ->, "$36^\circ$", angle radius=1.0cm, angle eccentricity=1.4]
		{angle = P_5--P_2--P_1};

	\pic[draw, ->, "$36^\circ$", angle radius=1.0cm, angle eccentricity=1.4]
		{angle = P_1--P_5--P_2};

	\pic[draw, ->, "$36^\circ$", angle radius=1.0cm, angle eccentricity=1.4]
		{angle = P_2--P_1--P_3};

	\pic[draw, ->, "$36^\circ$", angle radius=1.2cm, angle eccentricity=1.2]
		{angle = P_3--P_1--P_4};

	\pic[draw, ->, "$36^\circ$", angle radius=1.0cm, angle eccentricity=1.4]
		{angle = P_4--P_1--P_5};

	\pic[draw, ->, "$72^\circ$", angle radius=0.8cm, angle eccentricity=1.4]
		{angle = P_1--S_5--P_2};

	\pic[draw, ->, "$108^\circ$", angle radius=0.7cm, angle eccentricity=1.4]
		{angle = P_5--S_5--P_1};

% closing %{{{1

\end{tikzpicture}
\end{document}
