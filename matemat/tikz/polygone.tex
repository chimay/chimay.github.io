% Minimal TikZ standalone example
\documentclass[tikz, border=1mm]{standalone}

\usepackage{amsmath}
\usepackage{tikz}

\usetikzlibrary{calc,angles,quotes}

\begin{document}

\begin{tikzpicture}[scale=0.7]

	\coordinate (P1) at (-2,4);
	\coordinate (P2) at (1,5);
	\coordinate (P3) at (3,7);
	\coordinate (P4) at (5,6);
	\coordinate (P5) at (7,3);
	\coordinate (P6) at (4,0);
	\coordinate (P7) at (0,0);

	\draw (P1) \foreach \i in {2,...,7} { -- (P\i) } -- cycle;

	\foreach \i in {1,...,7} { \fill (P\i) circle (0.7mm); }

	 \node[left] at (P1) {$P1$};
	 \node[above left] at (P2) {$P2$};
	 \node[above] at (P3) {$P3$};
	 \node[above right] at (P4) {$P4$};
	 \node[right] at (P5) {$P5$};
	 \node[below] at (P6) {$P6$};
	 \node[below] at (P7) {$P7$};
