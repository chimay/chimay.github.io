\begin{document}
\begin{tikzpicture}[scale=1.2]

% parameters %{{{2

	\def\len{4}
	\def\height{2}
	\def\ang{12}
	\def\gle{-12}

% vectors %{{{2

	\coordinate (u) at (\ang:1);
	\coordinate (v) at (\gle:1);
	\coordinate (w) at (90:1);

% coordinates %{{{2

	\coordinate (A) at (0,0);
	\coordinate (X1) at ($ (A) - {0.5*\len}*(u) $);
	\coordinate (X2) at ($ (A) + {2.0*\len}*(u) $);

	\coordinate (B) at ($\height*(w)$);
	\coordinate (Y1) at ($ (B) - {0.5*\len}*(v) $);
	\coordinate (Y2) at ($ (B) + {2.0*\len}*(v) $);

	\coordinate (Z1) at ($ (A) - {0.3*\len}*(w) $);
	\coordinate (Z2) at ($ (B) + {0.3*\len}*(w) $);

% intersections %{{{2

	\path[name path=lineA] (X1) -- (X2);
	\path[name path=lineB] (Y1) -- (Y2);
	\path[name intersections={of=lineA and lineB, by=P}];

% vectors from coordinates %{{{2

	\coordinate (AB) at ($ (B) - (A) $);

% unit vectors %{{{3

	\coordinate (ab) at ($ (0,0)!1cm!(AB) $);

% coordinates from vectors %{{{2

	\coordinate (W1) at ($ (P) - {0.5*\len}*(ab) $);
	\coordinate (W2) at ($ (P) + {0.5*\len}*(ab) $);

% vertices, points, dots %{{{2

	\fill (A) circle (0.4mm);
	\fill (B) circle (0.4mm);
	\fill (P) circle (0.4mm);

% segments %{{{2

	\draw (X1) -- (X2);
	\draw (Y1) -- (Y2);
	\draw (Z1) -- (Z2);
	\draw (Z1) -- (Z2);
	\draw (W1) -- (W2);

% points, dots, vertices labels %{{{2

	\node[below right] at (P) {$P$};

% segments, sides labels %{{{2

	\node[above] at ($ (X1)!0.9!(X2) $) {$a$};
	\node[above] at ($ (Y1)!0.9!(Y2) $) {$b$};
	\node[left] at ($ (Z1)!0.95!(Z2) $) {$c$};
	\node[left] at ($ (W1)!0.95!(W2) $) {$d$};

% angles labels %{{{2

	\pic[draw, ->, "$\alpha$", angle radius=0.8cm, angle eccentricity=1.3]
	{angle = X2--A--B};

	\pic[draw, ->, "$\beta$", angle radius=0.6cm, angle eccentricity=1.3]
	{angle = A--B--Y2};

	\pic[draw, ->, "$\gamma$", angle radius=1.7cm, angle eccentricity=1.2]
	{angle = B--P--A};

	\pic[draw, ->, "$\alpha$", angle radius=0.8cm, angle eccentricity=1.3]
	{angle = W2--P--B};

	\pic[draw, ->, "$\beta$", angle radius=0.8cm, angle eccentricity=1.3]
	{angle = A--P--W1};

% closing %{{{2

\end{tikzpicture}
