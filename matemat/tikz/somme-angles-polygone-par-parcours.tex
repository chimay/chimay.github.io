\begin{document}
\begin{tikzpicture}[scale=0.7]

% parameters %{{{1

	\tikzmath{
		\arrowlength = 2.5 ;
		\arrowratio = 0.6 ;
		\numsides = int(6) ;
	}

% coordinates %{{{1

	\coordinate (P_1) at (0,0);
	\coordinate (P_2) at (4,0);
	\coordinate (P_3) at (7,3);
	\coordinate (P_4) at (5,6);
	\coordinate (P_5) at (2,7);
	\coordinate (P_6) at (-2,4);

% vectors %{{{2

	\foreach \i
		[
			evaluate=\i as \j using { int(mod(\i - 1 + 1, \numsides) + 1) },
		]
		in {1,...,\numsides}
	{
		\coordinate (v_\j) at ($ (P_\j) - (P_\i) $);
	}

% unit vectors %{{{2

	\foreach \i in {1,...,\numsides}
		\coordinate (u_\i) at ($ (0,0)!1cm!(v_\i) $);

% arrow tips %{{{2

	% ---- fixed length

	\foreach \i in {1,...,\numsides}
		\coordinate (A_\i) at ($ (P_\i) + \arrowlength*(u_\i) $);

	% ---- proportional length

% 	\foreach \i in {1,...,\numsides}
% 		\coordinate (A_\i) at ($ (P_\i) + \arrowratio*(v_\i) $);

% points, dots, vertices %{{{1

	\foreach \i in {1,...,\numsides}
		\fill (P_\i) circle (0.7mm);

% segments, sides, lines %{{{1

	\draw (P_1) \foreach \i in {2,...,\numsides} { -- (P_\i) } -- cycle;

	% ---- arrows

	\draw[dashed, ->] (P_1) -- (A_1);
	\draw[dashed, ->] (P_2) -- (A_2);
	\draw[dashed, ->] (P_3) -- (A_3);
	\draw[dashed, ->] (P_4) -- (A_4);
	\draw[dashed, ->] (P_5) -- (A_5);
	\draw[dashed, ->] (P_6) -- (A_6);

% points, dots, vertices labels %{{{1

	\node[below left] at (P_1) {$P_1$};

	%\node[above right] at (P1) {$P_1$};
	%\node[right] at (P2) {$P_2$};
	%\node[below] at (P3) {$P_3$};
	%\node[below] at (P4) {$P_4$};
	%\node[left] at (P5) {$P_5$};
	%\node[above left] at (P6) {$P_6$};

	% ---- vectors labels

	\node[left] at ($(P_1)!0.8!(A_1)$) {$\vec{u}_1$};

% angles labels %{{{1

	\foreach \i
		[
			evaluate=\i as \j using { int(mod(\i - 1 + 1, \numsides) + 1) },
		]
		in {1,...,\numsides}
	{
		\pic[draw, ->, "$\alpha_\i$", angle radius=0.9cm, angle eccentricity=1.4]
			{angle = A_\i--P_\i--P_\j};
	}

	\foreach \i
		[
			evaluate=\i as \j using { int(mod(\i - 1 + 1, \numsides) + 1) },
			evaluate=\i as \k using { int(mod(\i - 1 + 2, \numsides) + 1) }
		]
		in {1,...,\numsides}
	{
		\pic[draw, ->, "$\beta_\i$", angle radius=0.7cm, angle eccentricity=1.4]
		{angle = P_\k--P_\j--P_\i};
	}

% closing %{{{1

\end{tikzpicture}
\end{document}
