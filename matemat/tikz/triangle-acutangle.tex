		\path[name path=line2] (B) -- (Y);
		\path[name intersections={of=line1 and line2, by=C}];
	\end{pgfinterruptboundingbox}

	\fill (A) circle (0.4mm);
	\fill (B) circle (0.4mm);
	\fill (C) circle (0.4mm);

	\draw (A) -- (B) -- (C) -- cycle;

% points, dots, vertices labels %{{{1

	\node[below left] at (A) {$A$};
	\node[below right] at (B) {$B$};
	\node[above] at (C) {$C$};

% angles labels %{{{1

	\pic[draw, ->, "$\alpha$", angle radius=0.8cm, angle eccentricity=1.5]
		{angle = B--A--C};

	\pic[draw, ->, "$\beta$", angle radius=0.8cm, angle eccentricity=1.5]
		{angle = C--B--A};

		{angle = A--C--B};
