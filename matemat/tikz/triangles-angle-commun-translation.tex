\begin{tikzpicture}[scale=0.9]

% parameters %{{{1

	\def\x{2}
	\def\y{-3}
	\def\ratio{1.7}

	\def\rotation{-20}

% coordinates %{{{1

	\coordinate (A) at (0,0);
	\coordinate (B) at (4,0);
	\coordinate (C) at (3,3);

	\draw[vector, dashed] (X) -- (A);

	\node[above left] at (A) {$A$};
	\node[below right] at (B) {$B$};
	\node[above] at (C) {$C$};

	\node[below left] at (X) {$X$};
	\node[below right] at (Y) {$Y$};
	\node[above] at (Z) {$Z$};

% sides labels %{{{1

	\pic[draw, ->, "$\alpha$", angle radius=0.8cm, angle eccentricity=1.3]

	{angle = Y--X--Z};
