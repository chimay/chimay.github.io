% parameters %{{{1

	\tikzmath{
		\len = 8.0 ;
		\alp = 40.0 ;
		\bet = 25.0 ;
		\sinalpha = sin(\alp) ;
		\sinbeta = sin(\bet) ;
		\cosalpha = cos(\alp) ;
		\cosbeta = cos(\bet) ;
		\sinsum = sin(\alp + \bet) ;
		\cossum = cos(\alp + \bet) ;
	}

% coordinates %{{{1

	% ------- triangle ABC for alpha

	\coordinate (A) at (0,0);
	\coordinate (B) at ({\len*\cosbeta*\cosalpha}, 0);
	\coordinate (C) at ({\len*\cosbeta*\cosalpha}, {\len*\cosbeta*\sinalpha});

	% ------- triangle ABD for beta (unit hypotenuse)

	\coordinate (D) at ({\len*\cossum}, {\len*\sinsum});

	% ------- engulfing rectangle ABEF

	\coordinate (E) at ({\len*\cosbeta*\cosalpha}, {\len*\sinsum});
	\coordinate (F) at (0, {\len*\sinsum});

% points, dots, vertices %{{{1

	\fill (A) circle (0.4mm);
	\fill (B) circle (0.4mm);
	\fill (C) circle (0.4mm);
	\fill (D) circle (0.4mm);
	\fill (E) circle (0.4mm);
	\fill (F) circle (0.4mm);

% segments, sides, lines %{{{1

	\draw (A) -- (B) -- (C) -- cycle;
	\draw (A) -- (D) -- (C);

	\draw (C) -- (E) -- (F) -- (A);

	%\draw (A) rectangle (E);

% points, dots, vertices labels %{{{1

	\node[below left] at (A) {$A$};
	\node[below right] at (B) {$B$};
	\node[right] at (C) {$C$};
	\node[above] at (D) {$D$};
	\node[above right] at (E) {$E$};
	\node[above left] at (F) {$F$};

	\node[above left] at ($(A)!0.7!(C)$) {$\cos\beta$};
	\node[below left] at ($(D)!0.5!(C)$) {$\sin\beta$};

	\node[below]  at ($(A)!0.5!(B)$) {$\cos\beta \cos \alpha$};
	\node[right]  at ($(C)!0.5!(B)$) {$\cos\beta \sin \alpha$};

	\node[right]  at ($(E)!0.5!(C)$) {$\sin\beta \cos \delta$};
	\node[above]  at ($(D)!0.5!(E)$) {$\sin\beta \sin \delta$};

	\node[above] at ($(F)!0.5!(D)$) {$\cos(\alpha + \beta)$};
	\node[left] at ($(A)!0.5!(F)$) {$\sin(\alpha + \beta)$};
