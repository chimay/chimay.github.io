% Minimal TikZ standalone example

\documentclass[tikz,border=10pt]{standalone}

\usepackage{tikz}
\usepackage{tkz-euclide}

\usetikzlibrary{angles,quotes,calc}

\begin{document}
\begin{tikzpicture}[scale=1,>=stealth]

	% parameters

	\def\len{8.0}
	\def\a{40}
	\def\b{25}

	% base triangle ABC for alpha (unit hypotenuse)

	\coordinate (A) at (0,0);
	\coordinate (B) at ({\len*cos(\b)*cos(\a)},{\len*cos(\b)*sin(\a)});
	\coordinate (C) at ({\len*cos(\b)*cos(\a)},0);

	% triangle ADB for beta

	\coordinate (D) at ({\len*cos(\a+\b},{\len*sin(\a+\b)});

	% engulfing rectangle

	\coordinate (E) at ({\len*cos(\b)*cos(\a)},{\len*sin(\a+\b)});
	\coordinate (F) at (0,{\len*sin(\a+\b)});

	% points

	\fill (A) circle (0.4mm);
	\fill (B) circle (0.4mm);
	\fill (C) circle (0.4mm);
	\fill (D) circle (0.4mm);
	\fill (E) circle (0.4mm);
	\fill (F) circle (0.4mm);

	% projections

	%\coordinate (G) at ({cos(\a)*cos(\b)},0);

	% triangle sides

	\draw (A) -- (B) -- (C) -- cycle;
	\draw (A) -- (D) -- (B);
	%\draw[dashed] (D) -- (G);

	% rectangle

	\draw (A) rectangle (E);

	% segments labels

	\node[left] at ($(A)!0.5!(D)$) {$1$};

	\node[above left] at ($(A)!0.7!(B)$) {$\cos\beta$};
	\node[below left] at ($(D)!0.5!(B)$) {$\sin\beta$};

	\node[below]  at ($(A)!0.5!(C)$) {$\cos\beta \cos \alpha$};
	\node[right]  at ($(B)!0.5!(C)$) {$\cos\beta \sin \alpha$};

	\node[right]  at ($(E)!0.5!(B)$) {$\sin\beta \cos \delta$};
	\node[above]  at ($(D)!0.5!(E)$) {$\sin\beta \sin \delta$};

	\node[above] at ($(F)!0.5!(D)$) {$\cos(\alpha+\beta)$};
	\node[left] at ($(A)!0.5!(F)$) {$\sin(\alpha+\beta)$};

	% angle labels

	\pic[draw, ->, "$\alpha$", angle radius=1cm, angle eccentricity=1.2]
	{angle = C--A--B};

	\pic[draw, ->, "$\beta$", angle radius=1.5cm, angle eccentricity=1.2]
	{angle = B--A--D};

	\pic[draw, ->, "$\alpha+\beta$", angle radius=2.5cm, angle eccentricity=1.2]
	{angle = C--A--D};

	\pic[draw, ->, "$\gamma$", angle radius=1.1cm, angle eccentricity=1.3]
	{angle = A--B--C};

	\pic[draw, ->, "$\delta$", angle radius=1.1cm, angle eccentricity=1.3]
	{angle = E--B--D};

	% right angles

	\tkzMarkRightAngle[size=0.05](D,B,A)
	\tkzMarkRightAngle[size=0.05](B,C,A)
	\tkzMarkRightAngle[size=0.05](D,E,B)
	\tkzMarkRightAngle[size=0.05](A,F,D)

\end{tikzpicture}
\end{document}
