instrumentOrgan = \new GrandStaff \with {
  instrumentName = "Organ"
  shortInstrumentName =  "org"
} <<
  \new Staff = "right" \with {
    midiInstrument = "drawbar organ"
    midiMinimumVolume = #1.0
    midiMaximumVolume = #1.0
  } { \clef treble \voiceOrganRight }

    midiInstrument = "drawbar organ"
